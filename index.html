-- Zombie Survival Game Script (ServerScriptService) -- This script handles zombie spawning, player stats, and basic survival mechanics local Players = game:GetService("Players") local RunService = game:GetService("RunService") local Debris = game:GetService("Debris") local ReplicatedStorage = game:GetService("ReplicatedStorage") -- Settings local ZOMBIE_SPAWN_TIME = 5 -- seconds between spawns local MAX_ZOMBIES = 10 -- maximum zombies at once local ZOMBIE_HEALTH = 100 local ZOMBIE_SPEED = 8 local POINTS_PER_KILL = 10 -- Variables local zombies = {} local spawnPoints = {} local lastSpawnTime = 0 local gameRunning = true -- Find all spawn points in workspace for _, part in pairs(workspace:GetChildren()) do if part.Name == "ZombieSpawn" then table.insert(spawnPoints, part) end end -- Create leaderstats when player joins Players.PlayerAdded:Connect(function(player) local leaderstats = Instance.new("Folder") leaderstats.Name = "leaderstats" leaderstats.Parent = player local kills = Instance.new("IntValue") kills.Name = "Kills" kills.Value = 0 kills.Parent = leaderstats local points = Instance.new("IntValue") points.Name = "Points" points.Value = 0 points.Parent = leaderstats local survived = Instance.new("IntValue") survived.Name = "Time Survived" survived.Value = 0 survived.Parent = leaderstats end) -- Function to create a zombie local function createZombie(spawnPosition) -- Create zombie model local zombie = Instance.new("Model") zombie.Name = "Zombie" zombie.Parent = workspace -- Create zombie parts local head = Instance.new("Part") head.Name = "Head" head.Size = Vector3.new(2, 1, 1) head.BrickColor = BrickColor.new("Bright green") head.TopSurface = Enum.SurfaceType.Smooth head.BottomSurface = Enum.SurfaceType.Smooth head.Parent = zombie local torso = Instance.new("Part") torso.Name = "Torso" torso.Size = Vector3.new(2, 2, 1) torso.BrickColor = BrickColor.new("Bright green") torso.TopSurface = Enum.SurfaceType.Smooth torso.BottomSurface = Enum.SurfaceType.Smooth torso.Parent = zombie -- Position zombie zombie:SetPrimaryPartCFrame(CFrame.new(spawnPosition + Vector3.new(0, 3, 0))) torso.CFrame = CFrame.new(spawnPosition + Vector3.new(0, 3, 0)) head.CFrame = torso.CFrame + Vector3.new(0, 1.5, 0) -- Add humanoid local humanoid = Instance.new("Humanoid") humanoid.MaxHealth = ZOMBIE_HEALTH humanoid.Health = ZOMBIE_HEALTH humanoid.WalkSpeed = ZOMBIE_SPEED humanoid.Parent = zombie -- Weld parts together local neck = Instance.new("WeldConstraint") neck.Part0 = torso neck.Part1 = head neck.Parent = torso -- Add zombie to tracking table.insert(zombies, zombie) -- Handle zombie death humanoid.Died:Connect(function() -- Find who killed the zombie local tag = zombie:FindFirstChild("creator") if tag and tag.Value then local player = tag.Value if player and player.leaderstats then player.leaderstats.Kills.Value = player.leaderstats.Kills.Value + 1 player.leaderstats.Points.Value = player.leaderstats.Points.Value + POINTS_PER_KILL end end -- Remove from tracking for i, v in pairs(zombies) do if v == zombie then table.remove(zombies, i) break end end -- Clean up zombie after 3 seconds Debris:AddItem(zombie, 3) end) -- Simple AI: chase nearest player spawn(function() while zombie.Parent and humanoid.Health > 0 do local nearestPlayer = nil local shortestDistance = math.huge for _, player in pairs(Players:GetPlayers()) do if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then local distance = (zombie.Torso.Position - player.Character.HumanoidRootPart.Position).Magnitude if distance < shortestDistance then shortestDistance = distance nearestPlayer = player end end end if nearestPlayer then humanoid:MoveTo(nearestPlayer.Character.HumanoidRootPart.Position) end wait(1) -- Update AI every second end end) return zombie end -- Main game loop RunService.Heartbeat:Connect(function() if not gameRunning then return end local currentTime = tick() -- Spawn zombies if currentTime - lastSpawnTime >= ZOMBIE_SPAWN_TIME and #zombies < MAX_ZOMBIES and #spawnPoints > 0 then local randomSpawn = spawnPoints[math.random(1, #spawnPoints)] createZombie(randomSpawn.Position) lastSpawnTime = currentTime end -- Update player survival time for _, player in pairs(Players:GetPlayers()) do if player.Character and player.Character:FindFirstChild("Humanoid") then if player.Character.Humanoid.Health > 0 then player.leaderstats["Time Survived"].Value = math.floor(currentTime - (player:GetAttribute("StartTime") or currentTime)) end end end end) -- Set start time for players Players.PlayerAdded:Connect(function(player) player:SetAttribute("StartTime", tick()) end) print("Zombie Survival Game loaded! Make sure you have parts named 'ZombieSpawn' in workspace.")
